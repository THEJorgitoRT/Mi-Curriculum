C++ • Operaciones básicas 

          // Programa en C++ para operaciones básicas
          #include <iostream>
          using namespace std;

          int main() {
              int opcion;
              float num1, num2, resultado;

              cout << "===== CALCULADORA BASICA =====" << endl;
              cout << "1. Suma" << endl;
              cout << "2. Resta" << endl;
              cout << "3. Multiplicacion" << endl;
              cout << "4. Division" << endl;
              cout << "Elige una opcion (1-4): ";
              cin >> opcion;

              cout << "Ingresa el primer numero: ";
              cin >> num1;
              cout << "Ingresa el segundo numero: ";
              cin >> num2;

              switch(opcion) {
                  case 1:
                      resultado = num1 + num2;
                      cout << "Resultado de la suma: " << resultado << endl;
                      break;
                  case 2:
                      resultado = num1 - num2;
                      cout << "Resultado de la resta: " << resultado << endl;
                      break;
                  case 3:
                      resultado = num1 * num2;
                      cout << "Resultado de la multiplicacion: " << resultado << endl;
                      break;
                  case 4:
                      if(num2 != 0)
                          resultado = num1 / num2;
                      else {
                          cout << "Error: No se puede dividir entre cero" << endl;
                          return 0;
                      }
                      cout << "Resultado de la division: " << resultado << endl;
                      break;
                  default:
                      cout << "Opcion no valida" << endl;
              }

              return 0;
          }
